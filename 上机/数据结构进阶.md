## 1. 并查集（Union-Find / Disjoint Set Union）

### 1.1 基本并查集：路径压缩 + 按秩合并

**核心思想**：用树结构维护集合，支持合并和查询两个元素是否在同一集合。路径压缩使 `find` 时直接连向根节点；按秩合并让矮树挂到高树下，保持平衡。

**时间复杂度**：均摊 O(alpha(n))，alpha 为反阿克曼函数，近似 O(1)

```cpp
const int MAXN = 100005;
int fa[MAXN], rnk[MAXN];

void init(int n) {
    for (int i = 1; i <= n; i++) fa[i] = i, rnk[i] = 0;
}

int find(int x) {
    return fa[x] == x ? x : fa[x] = find(fa[x]); // 路径压缩
}

bool unite(int x, int y) {
    int fx = find(x), fy = find(y);
    if (fx == fy) return false;
    // 按秩合并
    if (rnk[fx] < rnk[fy]) swap(fx, fy);
    fa[fy] = fx;
    if (rnk[fx] == rnk[fy]) rnk[fx]++;
    return true;
}

bool same(int x, int y) { return find(x) == find(y); }
```

### 1.2 带权并查集

**核心思想**：在并查集基础上维护节点到根的权值（距离/关系）。`find` 时路径压缩的同时更新权值；`unite` 时根据已知关系推导根之间的权值关系。

**经典应用**：食物链问题（三种关系：同类、吃、被吃）

```cpp
int fa[MAXN];
int d[MAXN]; // d[x] 表示 x 到 fa[x] 的权值（关系）

void init(int n) {
    for (int i = 1; i <= n; i++) fa[i] = i, d[i] = 0;
}

int find(int x) {
    if (fa[x] == x) return x;
    int root = find(fa[x]);
    d[x] = (d[x] + d[fa[x]]) % 3; // 以食物链为例，模 3
    fa[x] = root;
    return root;
}

// 合并：已知 x 和 y 的关系为 relation
void unite(int x, int y, int relation) {
    int fx = find(x), fy = find(y);
    if (fx == fy) return; // 已在同一集合，需检查一致性
    fa[fx] = fy;
    d[fx] = (relation + d[y] - d[x] % 3 + 3) % 3;
}
```

**经典例题**：食物链（洛谷 P2024 / NOI 2001）
- 三种动物 A、B、C，A 吃 B，B 吃 C，C 吃 A。给出若干断言，判断多少条是假话。用带权并查集维护节点间 mod 3 的关系。

## 2. 树状数组（Binary Indexed Tree / Fenwick Tree）

### 2.1 单点修改 + 区间查询

**核心思想**：利用 lowbit 将前缀和拆分为 O(log n) 个区间段，支持高效的单点修改和前缀和查询。`lowbit(x) = x & (-x)`。

**时间复杂度**：修改 O(log n)，查询 O(log n)

```cpp
int tree[MAXN], n;

int lowbit(int x) { return x & (-x); }

void update(int i, int delta) { // 单点加 delta
    for (; i <= n; i += lowbit(i))
        tree[i] += delta;
}

int query(int i) { // 查询前缀和 a[1..i]
    int sum = 0;
    for (; i > 0; i -= lowbit(i))
        sum += tree[i];
    return sum;
}

int query(int l, int r) { // 区间和 a[l..r]
    return query(r) - query(l - 1);
}
```

### 2.2 区间修改 + 单点查询

**核心思想**：维护差分数组的树状数组。对差分数组做单点修改等价于原数组区间修改；查询差分数组的前缀和等价于原数组单点查询。

```cpp
int tree[MAXN], n;

void update(int i, int delta) {
    for (; i <= n; i += lowbit(i))
        tree[i] += delta;
}

void rangeAdd(int l, int r, int val) { // 区间 [l,r] 加 val
    update(l, val);
    update(r + 1, -val);
}

int pointQuery(int i) { // 查询 a[i] 的值
    int sum = 0;
    for (; i > 0; i -= lowbit(i))
        sum += tree[i];
    return sum;
}
```

### 2.3 区间修改 + 区间查询

**核心思想**：维护两个树状数组 `B1` 和 `B2`。利用公式：前缀和 `sum(1..x) = (x+1) * B1(x) - B2(x)`。

```cpp
long long B1[MAXN], B2[MAXN];
int n;

void add(long long* b, int i, long long delta) {
    for (; i <= n; i += lowbit(i)) b[i] += delta;
}

void rangeAdd(int l, int r, long long val) {
    add(B1, l, val);       add(B1, r + 1, -val);
    add(B2, l, val * l);   add(B2, r + 1, -val * (r + 1));
}

long long prefixSum(int x) {
    long long s1 = 0, s2 = 0;
    for (int i = x; i > 0; i -= lowbit(i)) s1 += B1[i], s2 += B2[i];
    return (x + 1) * s1 - s2;
}

long long rangeSum(int l, int r) {
    return prefixSum(r) - prefixSum(l - 1);
}
```

**经典例题**：逆序对计数
- 从右往左遍历数组，每个元素 a[i] 查询树状数组中比 a[i] 小的元素个数（已插入的），然后将 a[i] 插入。需要离散化。

## 3. 线段树（Segment Tree）

### 3.1 基本线段树：区间修改与查询

**核心思想**：将区间递归分为左右子区间，每个节点维护对应区间的聚合信息（和/最值等）。支持区间修改和区间查询。

**时间复杂度**：建树 O(n)，修改和查询 O(log n)

### 3.2 懒标记（Lazy Propagation）

**核心思想**：区间修改时，若当前节点的区间完全被修改区间覆盖，直接打标记而不递归下传。只有在后续需要访问子节点时，才将标记下推（pushdown）。

```cpp
const int MAXN = 100005;
long long tree[MAXN * 4], lazy[MAXN * 4];
int a[MAXN];

void pushup(int node) {
    tree[node] = tree[node * 2] + tree[node * 2 + 1];
}

void pushdown(int node, int len) {
    if (lazy[node]) {
        tree[node * 2] += lazy[node] * (len - len / 2);
        tree[node * 2 + 1] += lazy[node] * (len / 2);
        lazy[node * 2] += lazy[node];
        lazy[node * 2 + 1] += lazy[node];
        lazy[node] = 0;
    }
}

void build(int node, int l, int r) {
    lazy[node] = 0;
    if (l == r) { tree[node] = a[l]; return; }
    int mid = (l + r) / 2;
    build(node * 2, l, mid);
    build(node * 2 + 1, mid + 1, r);
    pushup(node);
}

void update(int node, int l, int r, int ql, int qr, long long val) {
    if (ql <= l && r <= qr) {
        tree[node] += val * (r - l + 1);
        lazy[node] += val;
        return;
    }
    pushdown(node, r - l + 1);
    int mid = (l + r) / 2;
    if (ql <= mid) update(node * 2, l, mid, ql, qr, val);
    if (qr > mid) update(node * 2 + 1, mid + 1, r, ql, qr, val);
    pushup(node);
}

long long query(int node, int l, int r, int ql, int qr) {
    if (ql <= l && r <= qr) return tree[node];
    pushdown(node, r - l + 1);
    int mid = (l + r) / 2;
    long long res = 0;
    if (ql <= mid) res += query(node * 2, l, mid, ql, qr);
    if (qr > mid) res += query(node * 2 + 1, mid + 1, r, ql, qr);
    return res;
}

// 调用：build(1, 1, n); update(1, 1, n, l, r, val); query(1, 1, n, l, r);
```

### 3.3 区间最值线段树

```cpp
int tree[MAXN * 4];

void build(int node, int l, int r) {
    if (l == r) { tree[node] = a[l]; return; }
    int mid = (l + r) / 2;
    build(node * 2, l, mid);
    build(node * 2 + 1, mid + 1, r);
    tree[node] = max(tree[node * 2], tree[node * 2 + 1]);
}

void update(int node, int l, int r, int pos, int val) { // 单点修改
    if (l == r) { tree[node] = val; return; }
    int mid = (l + r) / 2;
    if (pos <= mid) update(node * 2, l, mid, pos, val);
    else update(node * 2 + 1, mid + 1, r, pos, val);
    tree[node] = max(tree[node * 2], tree[node * 2 + 1]);
}

int query(int node, int l, int r, int ql, int qr) { // 区间最大值
    if (ql <= l && r <= qr) return tree[node];
    int mid = (l + r) / 2, res = INT_MIN;
    if (ql <= mid) res = max(res, query(node * 2, l, mid, ql, qr));
    if (qr > mid) res = max(res, query(node * 2 + 1, mid + 1, r, ql, qr));
    return res;
}
```

**经典例题**：区间修改区间查询（洛谷 P3372 / P3373）
- P3372：区间加法 + 区间求和，用上面的 lazy 模板。
- P3373：同时含加法和乘法两种操作，需要维护两个 lazy 标记，先乘后加。

## 4. ST 表（Sparse Table）- 区间最值查询（RMQ）

**核心思想**：预处理 `st[i][j]` 表示从位置 i 开始、长度为 2^j 的区间的最值。查询 [l, r] 时用两个可能重叠的区间覆盖，利用 max/min 的幂等性得到答案。

**时间复杂度**：预处理 O(n log n)，查询 O(1)

**适用范围**：静态数组的区间最值查询（不支持修改）。

```cpp
const int MAXN = 100005, LOG = 17;
int st[MAXN][LOG], lg[MAXN];

void buildST(int a[], int n) {
    lg[1] = 0;
    for (int i = 2; i <= n; i++) lg[i] = lg[i / 2] + 1;
    for (int i = 1; i <= n; i++) st[i][0] = a[i];
    for (int j = 1; j < LOG; j++)
        for (int i = 1; i + (1 << j) - 1 <= n; i++)
            st[i][j] = max(st[i][j-1], st[i + (1 << (j-1))][j-1]);
}

int queryMax(int l, int r) {
    int k = lg[r - l + 1];
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}
```

**经典例题**：静态区间最值（洛谷 P3865）
- 给定数组，多次查询区间最大值。ST 表是最优选择，O(1) 查询。

**注意**：ST 表只适用于满足"可重复贡献"性质的运算（max、min、gcd、or、and），不适用于区间求和。

## 5. 单调栈（Monotone Stack）

### 5.1 下一个更大元素

**核心思想**：维护一个栈，栈内元素从底到顶单调递减。遍历数组时，若当前元素大于栈顶，则栈顶的"下一个更大元素"就是当前元素，弹出栈顶。

**时间复杂度**：O(n)，每个元素至多入栈出栈各一次

```cpp
// 求每个位置右边第一个比它大的元素的下标，不存在则为 -1
vector<int> nextGreater(vector<int>& a) {
    int n = a.size();
    vector<int> res(n, -1);
    stack<int> stk; // 存下标
    for (int i = 0; i < n; i++) {
        while (!stk.empty() && a[stk.top()] < a[i]) {
            res[stk.top()] = i;
            stk.pop();
        }
        stk.push(i);
    }
    return res;
}
```

### 5.2 直方图最大矩形

**核心思想**：对每个柱子，找到它左右两侧第一个比它矮的柱子位置。以该柱子高度为矩形高度，宽度为左右边界之差减 1。用单调递增栈实现。

**时间复杂度**：O(n)

```cpp
// LeetCode 84: Largest Rectangle in Histogram
int largestRectangleArea(vector<int>& heights) {
    int n = heights.size();
    stack<int> stk;
    int maxArea = 0;
    for (int i = 0; i <= n; i++) {
        int h = (i == n) ? 0 : heights[i];
        while (!stk.empty() && heights[stk.top()] > h) {
            int height = heights[stk.top()]; stk.pop();
            int width = stk.empty() ? i : i - stk.top() - 1;
            maxArea = max(maxArea, height * width);
        }
        stk.push(i);
    }
    return maxArea;
}
```

**经典例题**：
- **下一个更大元素**（LeetCode 496 / 503）：直接使用单调栈。
- **直方图最大矩形**（LeetCode 84）：单调递增栈，是单调栈的经典应用。
- **接雨水**（LeetCode 42）：也可用单调递减栈求解。

## 6. 单调队列（Monotone Deque）

### 滑动窗口最大值

**核心思想**：维护一个双端队列，队列中存储下标，对应的值从前到后单调递减。每次滑动窗口时：
1. 弹出队首过期元素（下标超出窗口范围）。
2. 从队尾弹出所有不大于新元素的值。
3. 新元素入队尾。
4. 队首即为当前窗口最大值。

**时间复杂度**：O(n)

```cpp
// 滑动窗口大小为 k，求每个窗口的最大值
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
    deque<int> dq; // 存下标
    vector<int> res;
    for (int i = 0; i < (int)nums.size(); i++) {
        // 弹出过期元素
        while (!dq.empty() && dq.front() <= i - k)
            dq.pop_front();
        // 维护单调递减
        while (!dq.empty() && nums[dq.back()] <= nums[i])
            dq.pop_back();
        dq.push_back(i);
        if (i >= k - 1)
            res.push_back(nums[dq.front()]);
    }
    return res;
}
```

**经典例题**：滑动窗口最大值（LeetCode 239 / 洛谷 P1886）
- 给定数组和窗口大小 k，输出每个窗口的最大值。
- 同时求最大值和最小值：分别维护一个单调递减队列和一个单调递增队列。

**扩展应用**：
- 多重背包的单调队列优化
- DP 中的滑动窗口转移优化

## 7. 字典树（Trie）

### 7.1 字符串检索

**核心思想**：多叉树结构，每条边对应一个字符。从根到节点的路径表示一个字符串前缀。可以高效地插入和查询字符串。

**时间复杂度**：插入/查询 O(L)，L 为字符串长度

```cpp
const int MAXN = 100005 * 26; // 总节点数上限
int ch[MAXN][26], cntNode = 0;
bool isEnd[MAXN];

void init() {
    memset(ch[0], 0, sizeof(ch[0]));
    isEnd[0] = false;
    cntNode = 0;
}

void insert(const string& s) {
    int u = 0;
    for (char c : s) {
        int v = c - 'a';
        if (!ch[u][v]) {
            ch[u][v] = ++cntNode;
            memset(ch[cntNode], 0, sizeof(ch[cntNode]));
            isEnd[cntNode] = false;
        }
        u = ch[u][v];
    }
    isEnd[u] = true;
}

bool search(const string& s) {
    int u = 0;
    for (char c : s) {
        int v = c - 'a';
        if (!ch[u][v]) return false;
        u = ch[u][v];
    }
    return isEnd[u];
}

bool startsWith(const string& prefix) {
    int u = 0;
    for (char c : prefix) {
        int v = c - 'a';
        if (!ch[u][v]) return false;
        u = ch[u][v];
    }
    return true;
}
```

### 7.2 异或最大值（01-Trie）

**核心思想**：将整数的二进制表示插入 Trie（高位在前）。查询与给定数 x 异或最大的值时，从高位开始贪心选择与 x 当前位不同的分支。

**时间复杂度**：O(30) 或 O(log MAX_VAL) 每次操作

```cpp
int ch[MAXN * 32][2], cntNode = 0;

void insert(int x) {
    int u = 0;
    for (int i = 30; i >= 0; i--) {
        int bit = (x >> i) & 1;
        if (!ch[u][bit]) {
            ch[u][bit] = ++cntNode;
            ch[cntNode][0] = ch[cntNode][1] = 0;
        }
        u = ch[u][bit];
    }
}

int queryMaxXor(int x) {
    int u = 0, res = 0;
    for (int i = 30; i >= 0; i--) {
        int bit = (x >> i) & 1;
        int want = 1 - bit; // 贪心选择相反的位
        if (ch[u][want]) {
            res |= (1 << i);
            u = ch[u][want];
        } else {
            u = ch[u][bit];
        }
    }
    return res;
}
```

**经典例题**：
- **实现 Trie**（LeetCode 208）：标准字符串 Trie 的插入、搜索和前缀查询。
- **最大异或对**（洛谷 P4551）：给定 n 个数，求两个数异或的最大值。对每个数插入 01-Trie，然后对每个数查询能得到的最大异或值。

## 8. 哈希（Hash）

### 8.1 字符串哈希

**核心思想**：将字符串映射为一个整数（哈希值），使得字符串比较可以 O(1) 完成。采用多项式哈希：`h(s) = s[0] * p^(n-1) + s[1] * p^(n-2) + ... + s[n-1]`，其中 p 为基数，运算在模大素数下进行。

**时间复杂度**：预处理 O(n)，查询子串哈希 O(1)

```cpp
typedef unsigned long long ull;
const ull BASE = 131; // 或 13331
const ull MOD = 1e9 + 7;

ull h[MAXN], pw[MAXN]; // h[i] 为前缀哈希，pw[i] = BASE^i

void buildHash(const string& s) {
    int n = s.size();
    pw[0] = 1;
    for (int i = 1; i <= n; i++) pw[i] = pw[i-1] * BASE % MOD;
    h[0] = 0;
    for (int i = 0; i < n; i++)
        h[i+1] = (h[i] * BASE + s[i]) % MOD;
}

ull getHash(int l, int r) { // 获取 s[l..r] 的哈希值 (0-indexed)
    return (h[r+1] - h[l] * pw[r - l + 1] % MOD + MOD) % MOD;
}
```

### 8.2 Rabin-Karp 字符串匹配

**核心思想**：利用字符串哈希进行模式匹配。对模式串求哈希值，然后在文本串上滑动窗口，O(1) 比较哈希值。哈希相同再逐字符验证（防止冲突）。

**时间复杂度**：平均 O(n + m)，最坏 O(nm)（冲突多时）

```cpp
vector<int> rabinKarp(const string& text, const string& pattern) {
    int n = text.size(), m = pattern.size();
    if (m > n) return {};
    
    ull patHash = 0, txtHash = 0;
    ull power = 1;
    for (int i = 0; i < m - 1; i++) power = power * BASE % MOD;
    
    for (int i = 0; i < m; i++) {
        patHash = (patHash * BASE + pattern[i]) % MOD;
        txtHash = (txtHash * BASE + text[i]) % MOD;
    }
    
    vector<int> res;
    for (int i = 0; i <= n - m; i++) {
        if (txtHash == patHash && text.substr(i, m) == pattern) {
            res.push_back(i);
        }
        if (i + m < n) {
            txtHash = (txtHash - text[i] * power % MOD + MOD) % MOD;
            txtHash = (txtHash * BASE + text[i + m]) % MOD;
        }
    }
    return res;
}
```

### 8.3 双哈希防冲突

**核心思想**：使用两组不同的 (BASE, MOD) 参数计算哈希值，只有两个哈希值都相同时才认为字符串相等。冲突概率降低到约 1/(MOD1 * MOD2)。

```cpp
const ull BASE1 = 131, MOD1 = 1e9 + 7;
const ull BASE2 = 13331, MOD2 = 1e9 + 9;

struct DoubleHash {
    ull h1[MAXN], h2[MAXN], pw1[MAXN], pw2[MAXN];
    
    void build(const string& s) {
        int n = s.size();
        pw1[0] = pw2[0] = 1;
        h1[0] = h2[0] = 0;
        for (int i = 0; i < n; i++) {
            pw1[i+1] = pw1[i] * BASE1 % MOD1;
            pw2[i+1] = pw2[i] * BASE2 % MOD2;
            h1[i+1] = (h1[i] * BASE1 + s[i]) % MOD1;
            h2[i+1] = (h2[i] * BASE2 + s[i]) % MOD2;
        }
    }
    
    pair<ull,ull> get(int l, int r) { // 0-indexed [l, r]
        ull v1 = (h1[r+1] - h1[l] * pw1[r-l+1] % MOD1 + MOD1) % MOD1;
        ull v2 = (h2[r+1] - h2[l] * pw2[r-l+1] % MOD2 + MOD2) % MOD2;
        return {v1, v2};
    }
};
```

**经典例题**：
- **判断两子串是否相等**：预处理字符串哈希，O(1) 比较任意两个子串。
- **最长公共子串**：二分答案长度 + 哈希判断，O(n log n)。
- **最长回文子串**：二分 + 哈希判断正反串是否相同。

## 总结

| 数据结构 | 核心操作 | 时间复杂度 | 适用场景 |
|---------|---------|-----------|---------|
| 并查集 | 合并、查询 | 均摊 O(alpha(n)) | 连通性、集合合并 |
| 树状数组 | 单点/区间修改、前缀和查询 | O(log n) | 动态前缀和、逆序对 |
| 线段树 | 区间修改、区间查询 | O(log n) | 区间聚合操作、带 lazy |
| ST 表 | 区间最值查询 | 预处理 O(n log n)，查询 O(1) | 静态 RMQ |
| 单调栈 | 寻找下一个更大/更小元素 | O(n) | NGE、直方图最大矩形 |
| 单调队列 | 滑动窗口最值 | O(n) | 滑动窗口、DP 优化 |
| Trie | 字符串插入/查询 | O(L) | 字符串检索、异或最值 |
| 字符串哈希 | 子串哈希比较 | 预处理 O(n)，查询 O(1) | 子串判等、模式匹配 |

## 经典例题精讲

### 例题 1：LC 739 - 每日温度

**题意**：给定每日温度数组 temperatures，对每一天求需要等待几天才能等到更高温度。若之后不会更高则为 0。

**思路**：单调递减栈，栈中存下标。遍历数组，当前温度高于栈顶时弹出栈顶并计算等待天数。

```cpp
class Solution {
public:
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> ans(n, 0);
        stack<int> stk; // 存下标，对应温度单调递减
        for (int i = 0; i < n; i++) {
            while (!stk.empty() && temperatures[stk.top()] < temperatures[i]) {
                int j = stk.top(); stk.pop();
                ans[j] = i - j;
            }
            stk.push(i);
        }
        return ans;
    }
};
```

### 例题 2：LC 307 - 区域和检索 - 数组可修改

**题意**：实现一个数据结构，支持单点修改和区间求和。

**思路**：树状数组模板。update 时对差值做单点加，sumRange 用前缀和相减。

```cpp
class NumArray {
    vector<int> tree;
    vector<int> nums;
    int n;

    int lowbit(int x) { return x & (-x); }

    void add(int i, int delta) {
        for (; i <= n; i += lowbit(i))
            tree[i] += delta;
    }

    int prefixSum(int i) {
        int s = 0;
        for (; i > 0; i -= lowbit(i))
            s += tree[i];
        return s;
    }

public:
    NumArray(vector<int>& nums) : nums(nums), n(nums.size()), tree(nums.size() + 1, 0) {
        for (int i = 0; i < n; i++)
            add(i + 1, nums[i]);
    }

    void update(int index, int val) {
        add(index + 1, val - nums[index]);
        nums[index] = val;
    }

    int sumRange(int left, int right) {
        return prefixSum(right + 1) - prefixSum(left);
    }
};
```

### 例题 3：LC 347 - 前K个高频元素

**题意**：给定整数数组 nums 和整数 k，返回出现频率前 k 高的元素，顺序不限。

**思路**：哈希表统计频率，然后用大小为 k 的小根堆维护前 k 大。也可用桶排序做到 O(n)。

```cpp
class Solution {
public:
    vector<int> topKFrequent(vector<int>& nums, int k) {
        unordered_map<int, int> freq;
        for (int x : nums) freq[x]++;

        // 小根堆，按频率排序，堆大小保持 k
        auto cmp = [](pair<int,int>& a, pair<int,int>& b) {
            return a.second > b.second;
        };
        priority_queue<pair<int,int>, vector<pair<int,int>>, decltype(cmp)> pq(cmp);

        for (auto& [val, cnt] : freq) {
            pq.push({val, cnt});
            if ((int)pq.size() > k) pq.pop();
        }

        vector<int> ans;
        while (!pq.empty()) {
            ans.push_back(pq.top().first);
            pq.pop();
        }
        return ans;
    }
};
